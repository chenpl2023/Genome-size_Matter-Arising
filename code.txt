####Raw data quality control

 for filename in *_1.fastq ; do base=$(basename $filename _1.fastq); echo$base; fastp -i ${base}_1.fastq -I ${base}_2.fastq -o ${base}_R1_clean.fastq -O ${base}_R2_clean.fastq --thread 8; done

#####Sequence assembly
ls *_R1_clean.fastq | while read id
do
id=${id/_R1_clean.fastq/}
megahit -t 60 -m 0.3 -1 ${id}_R1_clean.fastq -2 ${id}_R2_clean.fastq -o /data2/Chenpl/PRJNA1018220/${id} --out-prefix ${id} --k-list 29,39,59,79,99,119,141
done

#####Remove eukaryotic-related sequences#####
conda activate EukRep
ls *.contigs.fa | while read id
do
id=${id/.contigs.fa/}
EukRep -i ${id}.contigs.fa -o contigs.euk/${id}.euk.contigs.fa --prokarya contigs.prok/${id}.prok.contigs.fa --min 200 
done

#####Construct non-redundant contig set
chmod 755 ChangeFastaName.py

ls *.prok.contigs.fa | while read id
do
id=${id/.prok.contigs.fa/}
./ChangeFastaName.py ${id}.prok.contigs.fa ${id} 0 > ${id}.re.prok.contigs.fa
done

cat *.re.prok.contigs.fa >> all200_contigs.fa &

conda activate redundans_2.0.1

nohup redundans.py -v -f all200_contigs.fa -o all200_contigs.reduced.fa -t 80 --log log --overlap 0.80 --identity 0.90 --noscaffolding --nogapclosing &

#Identify viral sequences
conda activate vibrant
nohup VIBRANT_run.py -i all_contigs.reduced.fa -folder vibrant_out -t 30 &

#Remove virus-related sequences
ls *.contigs.fa | while read id
do
id=${id/.contigs.fa/}
nohup bbduk.sh in=${id}.contigs.fa out=${id}.contigs.novirus.fa ref=all200contigs.reduced.phages_combined.ffn
done


#####Calculate the average prokaryotic genome size

conda activate genomeSize

ls *.prok.contigs.novirus.fa | while read id
do
id=${id/.prok.contigs.novirus.fa/}
nohup run_microbe_census.py -t 20 ${id}.prok.contigs.novirus.fa genomeSize/${id}.txt
done


ls *.prok.contigs.fa | while read id
do
id=${id/.prok.contigs.fa/}
./ChangeFastaName.py ${id}.prok.contigs.fa ${id} 0 > ${id}.re.prok.contigs.fa
done

cat *.re.prok.contigs.fa >> all200_contigs_new.fa &

#####Contig annotation
nohup kraken2 --report report1.txt --threads 30 --db /data1/kraken2_DB_core_NT all200_contigs_new.fa > all200_contigs_krakenclassified1.txt &

 #Isolate bacterial-related sequences
ls *.re.prok.contigs.fa | while read id
do
id=${id/.re.prok.contigs.fa/}
extract_kraken_reads.py -k all200_contigs_krakenclassified1.txt -s ${id}.re.prok.contigs.fa -o ${id}.Kraken_bacteria.fa  --taxid 2  --include-children  --report report1.txt
done

 # Isolate archaea-related sequences
ls *.re.prok.contigs.fa | while read id
do
id=${id/.re.prok.contigs.fa/}
extract_kraken_reads.py -k all200_contigs_krakenclassified1.txt -s ${id}.re.prok.contigs.fa -o ${id}.Kraken_Archaea.fa  --taxid 2157 --include-children  --report report1.txt
done


conda activate genomeSize
#2.2 Calculate the average  genome size
ls *.Kraken_bacteria.fa | while read id
do
id=${id/.Kraken_bacteria.fa/}
nohup run_microbe_census.py -t 20 ${id}.Kraken_bacteria.fa genomeSize/${id}.txt
done

ls *.Kraken_Archaea.fa | while read id
do
id=${id/.Kraken_Archaea.fa/}
nohup run_microbe_census.py -t 20 ${id}.Kraken_Archaea.fa genomeSize/${id}.txt
done
